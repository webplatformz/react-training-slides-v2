Besides rendering, React provides a basic state management functionality.

useState

                
import { useState } from "react";

function MyComponent() {
  const [name, setName] = useState();

  return <>
    <p>
      Enter your name:
      <input
        type="text"
        onChange={(e) => setName(e.target.value)} />
    </p>

    <h1>Hello, {name}</h1>
  </>;
}
